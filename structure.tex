% Each refsection has its own counter prefix
\newcommand{\refsectionletter}[1]{%
    \ifcase#1\relax%
    \or P%
    \or C%
    \or F%
    \or O%
    \else A%
    \fi%
}
\AtDataInput{%
  \ifkeyword{Attended}
    {}
    {\csnumgdef{entrycount:\therefsection}{%
      \csuse{entrycount:\therefsection}+1}}}
% Print the labelnumber as the total number of entries in the
% current refsection, minus the actual labelnumber, plus one
\DeclareFieldFormat{labelnumber}{\refsectionletter{\therefsection}\mkbibdesc{#1}}    
\newrobustcmd*{\mkbibdesc}[1]{%
  \number\numexpr\csuse{entrycount:\therefsection}+1-#1\relax}


% Sort by year, month, day, title
\DeclareSortingScheme{ymdnt}{
  \sort{
    \field{presort}
  }
  \sort[final]{
    \field{sortkey}
  }
  \sort[direction=descending]{
    \field[strside=left,strwidth=4]{sortyear}
    \field[strside=left,strwidth=4]{year}
    \literal{9999}
  }
  \sort[direction=descending]{
    \field{month}
    \literal{9999}
  }
  \sort{
    \field{sortname}
    \field{author}
    \field{editor}
    \field{translator}
    \field{sorttitle}
    \field{title}
  }
  \sort{
    \field{sorttitle}
    \field{title}
  }
}

\renewbibmacro*{journal+issuetitle}{%
  \iffieldundef{journaltitle}
    {}
    {\printtext[journaltitle]{%
       \printfield[titlecase]{journaltitle}}}%
  \setunit*{\addspace}%
  \printfield{volume}%
  \setunit*{\addcomma\space}%
  \printfield{pages}%
  \setunit*{\addspace}%
  \printtext{(}%
  \printfield{year}%
  \printtext{)}%
}

\renewbibmacro*{doi+eprint+url}{%
  \iffieldundef{doi}
    {}
    {\printfield{doi}}%
}

\DeclareFieldFormat{journallink}{%
  \iffieldundef{doi}
    {#1}
    {\href{https://doi.org/\thefield{doi}}{\textcolor{my_green_d}{#1}}}%
}

\DeclareFieldFormat{arxivlink}{\href{https://arxiv.org/abs/#1}{\textcolor{my_red_d}{arXiv:#1}}}


\DeclareFieldFormat{pages}{#1}

\DeclareFieldFormat{citations}{
  \iffieldundef{doi}
    {Citations: #1}
    {\href{https://badge.dimensions.ai/details/doi/\thefield{doi}}{Citations: #1}}%
}



\DeclareBibliographyDriver{article}{%
  % Row 1: title
  \printfield{title}%
  \newline
  % Row 2: authors
  \printnames{author}%
  \newline
  % Row 3: journal reference in green, arXiv in red
  \printtext[journallink]{%
    \iffieldundef{shortjournal}
      {\printfield{journaltitle}}
      {\printfield{shortjournal}}%
    \setunit*{\addcomma\space}%
    \printfield{volume}%
    \setunit*{\addcomma\space}%
    \printfield{pages}%
    \setunit*{\addspace}%
    \printtext{(}\printfield{year}\printtext{)}%
  }%
  \iffieldundef{eprint}
    {}
    {\addcomma\addspace\printfield[arxivlink]{eprint}}%
  % Row 4: citation count
  \iffieldundef{citations}
    {}
    {\ifnumgreater{\thefield{citations}}{0}
      {\newline\printfield{citations}}
      {}}%
  \finentry
}

\DeclareFieldFormat{noteyearlink}{%
  \iffieldundef{doi}
    {#1}
    {\href{https://doi.org/\thefield{doi}}{\textcolor{my_red_d}{#1}}}%
}

\DeclareBibliographyDriver{misc}{%
  % Row 1: title
  \printfield{title}%
  \newline
  % Row 2: authors
  \printnames{author}%
  % Row 3: omitted for preprints
  % Row 4: arXiv reference in red (via note field)
  \newline
  \printfield[noteyearlink]{note}%
  \finentry
}


\DeclareBibliographyDriver{unpublished}{%
  % Row 1: Bold date. Conference name.
  \textbf{%
    \iffieldundef{month}
      {\printfield{year}}
      {\mkbibmonth{\thefield{month}}\addnbspace\printfield{year}}%
  }%
  \printtext{. }%
  \printfield{title}%
  \newline
  % Row 2: Venue
  \printfield{venue}%
  % Row 3: contribution type + italic title (only if type is defined)
  \iffieldundef{type}
    {}
    {\newline\printfield{type}\addspace\textit{\printfield{note}}}%
  \finentry
}

  \usebibmacro{pi_macro}

    {\usebibmacro{related:init}%
     \usebibmacro{related}}
    {}%
  \usebibmacro{finentry}}

  \newbibmacro*{title+institution}{
    \printfield{title}
    \printtext{at}
    \printfield{institution}
  }
  \newbibmacro*{agency+code+period}{
    {\printtext[parens]{
        \printfield{agency}
        \setunit*{\addcomma\space}
        \printfield{code}
        \printtext{for the period}
        \printfield{period}
    }}
  \newunit}

    \newunit
    \ifboolexpr{
        test {\iffieldundef{value}}
    }
    {}
    {
    \printtext{Ammount:}
    \printfield{value}
    }
  }

    \ifboolexpr{
        test {\iffieldundef{pis}}

    {}

    }
    }
